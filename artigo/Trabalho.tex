\section{Definições}

\begin{defi}
 Seja uma função $S:\mathcal{F}(X)x\mathcal{F}(X) \rightarrow [0,1]$. $S(A,B)$ é uma medida de subsethood se para $A,B,C \in \mathcal{F}(X)$ temos:
 \begin{enumerate}
 \item $S(A,B)=1 \Leftrightarrow A \subseteq B$ 
 \item $S(X,\emptyset)=0$ 
 \item Se $A \subseteq B \subseteq C$, então $S(C,A) \leq S(B,A)$ e $S(C,A) \leq S(C,B)$
\end{enumerate}
\end{defi}

\begin{defi}
 A medida de subsethood de Kosko é definida da seguinte forma:
 \begin{center}
 $S_k(A,B)=\frac{\int_{x \in X}min(\mu_A(x),\mu_B(x))}{\int_{x \in X}\mu_A(x)}$ 
 \end{center}
\end{defi}

\begin{defi}
 A medida de subsethood de Willmott é definida da seguinte forma:
\begin{center}
 $S_w(A,B)=\frac{\int_{x \in X}\mu_B(x)}{\int_{x \in X}max(\mu_A(x),\mu_B(x))}$
\end{center}
\end{defi}

\begin{defi}
 As medidas de subsethood $S_p^{\cap}$(meet) e $S_p^{\cup}$(join) são definidas da seguinte forma:
\begin{center}
 $S_p^{\cap}(A,B)=I_p(v_p(A),v_p(A \cap B))$,
 \\$S_p^{\cup}(A,B)=I_p(v_p(A \cup B),v_p(B))$
\end{center}
 Onde $I_p$ denota a implicação de Goguen e $v_p$ é uma função $\mathcal{F}(X) \rightarrow [0,1]$ definida por:
\begin{center}
 $v_p=\sum_{i=1}^{k}\frac{1-cos(\pi[\mu_C(x^i)]^p)}{k}, \quad X=\{x^1,...,x^k\}$
\end{center}
\end{defi}
