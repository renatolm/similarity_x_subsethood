\section{Definições}

\begin{defi}
 Seja uma função $S:\mathcal{F}(X)x\mathcal{F}(X) \rightarrow [0,1]$. $S(A,B)$ é uma medida de subsethood se para $A,B,C \in \mathcal{F}(X)$ temos:
 \begin{enumerate}
 \item $S(A,B)=1 \Leftrightarrow A \subseteq B$ 
 \item $S(X,\emptyset)=0$ 
 \item Se $A \subseteq B \subseteq C$, então $S(C,A) \leq S(B,A)$ e $S(C,A) \leq S(C,B)$
\end{enumerate}
\end{defi}

\begin{defi}
 A medida de subsethood de Kosko é definida da seguinte forma:
 \begin{center}
 $S_k(A,B)=\frac{\int_{x \in X}min(\mu_A(x),\mu_B(x))}{\int_{x \in X}\mu_A(x)}$ 
 \end{center}
\end{defi}

\begin{defi}
 A medida de subsethood de Willmott é definida da seguinte forma:
\begin{center}
 $S_w(A,B)=\frac{\int_{x \in X}\mu_B(x)}{\int_{x \in X}max(\mu_A(x),\mu_B(x))}$
\end{center}
\end{defi}

\begin{defi}
 As medidas de subsethood $S_p^{\cap}$(meet) e $S_p^{\cup}$(join) são definidas da seguinte forma:
\begin{center}
 $S_p^{\cap}(A,B)=I_p(v_p(A),v_p(A \cap B))$,
 \\$S_p^{\cup}(A,B)=I_p(v_p(A \cup B),v_p(B))$
\end{center}
 Onde $I_p$ denota a implicação de Goguen
 \begin{center}
 \[
 I_p(x,y) = 
  \begin{cases} 
   1 & \text{se } x = 0 \\
   \min(1,\frac{y}{x})       & \text{se } x > 0
  \end{cases}
\]
\end{center}
  e $v_p$ é uma função $\mathcal{F}(X) \rightarrow [0,1]$ definida por:
\begin{center}
 $v_p=\sum_{i=1}^{k}\frac{1-cos(\pi[\mu_C(x^i)]^p)}{k}, \quad X=\{x^1,...,x^k\}$
\end{center}
\end{defi}

\subsection{Comentários}
\begin{itemize}
\item Em todos os casos, as medidas de subsethood do tipo {\it meet} ($S_p^{\cap}$) com $p \geq 10$ não apresentam valores de ativação significativos. No Caso 2, temos $S_{10}^{\cap}=0.323$ apesar de a entrada estar quase totalmente contida no antecedente. \ul{Portanto, não parece interessante considerar as medidas do tipo {\it meet} ($S_p^{\cap}$) com $p \geq 10$}.

\item Também é possível notar que conforme o parâmetro $p$ aumenta, o grau de ativação pela medida de subsethood do tipo {\it join} ($S_p^{\cup}$) tende a se igualar para todas as regras. \ul{Portanto, também não parece interessante considerar as medidas do tipo {\it join} ($S_p^{\cup}$) com $p \geq 10$}.

\item No Caso 1, apesar de a entrada apresentar um nível considerável de interseção com o antecedente da Regra 2 (tomate verde), as medidas de subsethood do tipo {\it join} ($S_p^{\cup}$) ainda apresentam um elevado grau de ativação, em geral $\geq 0.9$. \ul{Desta forma, é possível concluir que para entradas com cardinalidade pequena, deve-se dar um peso maior para o grau de ativação das medidas de subsethood do tipo {\it meet} ($S_p^{\cap}$)}.

\item Nos Casos 4 e 5, pode-se notar que o grau de ativação das regras independe da localização da entrada no caso em que a interseção da entrada com os antecedentes é vazia. A medida de subsethood do tipo {\it join} ($S_p^{\cup}$) considera apenas a cardinalidade dos antecedentes e da entrada, o que pode representar um problema em bases de regras esparsas.

\end{itemize}

\section{Experimentos com combinação de medidas}
 
\begin{prop}
A média ponderada entre medidas de subsethood dos tipos {\it meet} e {\it join}:
\begin{center}
$S_p(A,B)=\frac{\alpha S_p^{\cap}(A,B) + \beta S_p^{\cup}(A,B)}{\alpha + \beta}$
\end{center}
é uma medida de subsethood
\end{prop}

\begin{proof}
\begin{enumerate}
 \item $S(A,B)=1 \Leftrightarrow A \subseteq B$ \\
 \underline{$\Rightarrow$:}\quad $S_p(A,B)=1$ \\
 Suponha que $A \nsubseteq B \Rightarrow S_p^{\cap}(A,B) < 1$ e $S_p^{\cup}(A,B) < 1$ \\
 $\Rightarrow \alpha S_p^{\cap}(A,B) < \alpha$ e $\beta S_p^{\cup}(A,B) < \beta$ \\
 $\Rightarrow \alpha S_p^{\cap}(A,B) + \beta S_p^{\cup}(A,B) < \alpha + \beta$ \\
 $\Rightarrow S_p(A,B) < 1$, o que contradiz a hipótese inicial. \\
 Portanto, $A \subseteq B$. \\
 \underline{$\Leftarrow$:}\quad $A \subseteq B$ \\
 $\Rightarrow S_p^{\cap}(A,B) = 1$ e $S_p^{\cup}(A,B) = 1$ \\
 $\Rightarrow \alpha S_p^{\cap}(A,B) + \beta S_p^{\cup}(A,B) = \alpha + \beta$ \\
 $\Rightarrow S_p(A,B)=1$ \\
 \item $S(X,\emptyset)=0$ \\
 $S_p^{\cap}(X,\emptyset) = 0$ e $S_p^{\cup}(X,\emptyset) = 0$ \\
 $\Rightarrow S(X,\emptyset)=\frac{\alpha \cdot 0 + \beta \cdot 0}{\alpha + \beta} = 0$ \\
 \item Se $A \subseteq B \subseteq C$, então $S(C,A) \leq S(B,A)$ e $S(C,A) \leq S(C,B)$
\end{enumerate}
\end{proof}

\section{Modificadores de números fuzzy}

\begin{defi}
Um \textit{modificador fuzzy m} sobre X é uma aplicação do tipo:
\begin{center}
$m: \mathcal{F}(X) \rightarrow \mathcal{F}(X)$
\end{center}
tal que \textit{m} é dito:
\begin{enumerate}
\item \textit{Expansivo} se, para todo $A \in \mathcal{F}(X), A \subseteq m(A)$, ou seja, $\mu_{A}(x) \leq \mu_{m(A)}(x)$, $\forall x \in X$;
\item \textit{Restritivo} se, para todo $A \in \mathcal{F}(X), A \supseteq m(A)$, ou seja, $\mu_{A}(x) \geq \mu_{m(A)}(x)$, $\forall x \in X$;
\end{enumerate}
\end{defi}

\hfill

\begin{prop}
\label{procedimento}
Considere o seguinte procedimento para construção de números fuzzy modificados:
\begin{enumerate}
\item Centrar A em torno de 0;
\item Adicionar c a ambos os lados de $supp(A)$;
\item Calcular $f = \frac{supp(A)}{supp(A^c)}$;
\item Multiplicar todos os $\alpha$-cortes por f;
\item Transladar o resultado de volta.
\end{enumerate}
Isto deve funcionar para todos os números fuzzy, menos números fuzzy crisp, que podem 
ser vistos como números fuzzy triangulares. Para $r \in \mathds{R}$, podemos definir $r^c 
= (r-c;r;r+c)$.
\end{prop}

\begin{exe}
Seja o seguinte número fuzzy $A \in \mathcal{F}(\mathds{R})$:
\begin{figure}[H]
 \centering
  \includegraphics[width=0.8\textwidth]{numero_orig.png}
  \caption{Número fuzzy a ser modificado.}
  \label{fig:numero_orig}
\end{figure}
Com função de pertinência dada por:
\begin{center}
\[
 \mu_A(x) = 
  \begin{cases} 
   x-2 & \text{se } 2 < x \leq 3 \\
   4-x & \text{se } 3 < x \leq 4 \\
   0 & \text{c.c. }
  \end{cases}
\]
\end{center}
E $\alpha$-cortes dados por:
\begin{center}
$A^{\alpha} = [\alpha + 2, -\alpha + 4]$
\end{center}
Aplicando o procedimento descrito na Proposição \cite{procedimento}, temos:
\begin{itemize}
\item \textbf{Passo 1}: Centrar A em torno de 0.
\begin{figure}[H]
 \centering
  \includegraphics[width=0.8\textwidth]{numero_center.png}
  \caption{Número fuzzy centrado em 0.}
  \label{fig:numero_center}
\end{figure}

\item \textbf{Passo 2}: Adicionar c a ambos os lados de $supp(A')$.
Considerando c = 1, temos:
\begin{center}
$supp(A') = [-1,1] \Rightarrow supp(A'^c) = [-2,2] \Rightarrow supp(A^c) = [1,5]$
\end{center}

\item \textbf{Passo 3}: Calcular $f = \frac{supp(A)}{supp(A^c)}$.
\begin{align*}
f &= \frac{supp(A)}{supp(A^c)} = \frac{[2,4]}{[1,5]}
 = [2,4]\cdot [\frac{1}{5},1] \\
 &= [\min(2 \cdot \frac{1}{5},4 \cdot \frac{1}{5},2 \cdot 1,4 \cdot 1), \max(2 \cdot \frac{1}{5},4 \cdot \frac{1}{5},2 \cdot 1,4 \cdot 1)] \\
  &= [\frac{2}{5}, 4]
\end{align*}

\item \textbf{Passo 4}: Multiplicar todos os $\alpha$-cortes por f.
\begin{align*}
B^{\alpha} &= f \cdot A^{\alpha} = [\frac{2}{5}, 4] \cdot A^{\alpha} \\
 &= [\min(\frac{2}{5} \cdot (\alpha + 2),\frac{2}{5} \cdot (-\alpha + 4),4 \cdot (\alpha + 2),4 \cdot (-\alpha + 4)),\\ &\max(\frac{2}{5} \cdot (\alpha + 2),\frac{2}{5} \cdot (-\alpha + 4),4 \cdot (\alpha + 2),4 \cdot (-\alpha + 4))] \\
  &= [(\frac{2}{5} \cdot \alpha + \frac{4}{5}),(-4 \cdot \alpha + 16)]
\end{align*}
\begin{figure}[H]
 \centering
  \includegraphics[width=0.8\textwidth]{numero_resultado.png}
  \caption{Número fuzzy modificado.}
  \label{fig:numero_resultado}
\end{figure}
\end{itemize}
\end{exe}
